6 case class Grid(size: Int = 3) {
7     val eol: String = sys.props("line.separator") // End of line character
8 
9     // Initialize an empty grid with None values, representing empty spots
10     private var grid: Array[Array[Option[Card]]] = Array.fill(size, size)(None)
11 
12     // Helper method to check if a position is within the grid bounds
13     private def isWithinBounds(x: Int, y: Int): Boolean = {
14         x >= 0 && x < size && y >= 0 && y < size
15     }
16 
17     // Method to place a card on the grid
18     def placeCard(x: Int, y: Int, card: Card, rectangleColor: Suit): Boolean = {
19         if (isWithinBounds(x, y) && grid(x)(y).isEmpty) {
20             grid(x)(y) = Some(card)
21             true
22         } else {
23             false  // Return false if the position is out of bounds or already occupied
24         }
25     }
26 
27     // Method to calculate points for a given position
28     def calculatePoints(x: Int, y: Int, rectangleColor: Suit): Int = {
29         grid(x)(y) match {
30             case Some(card) =>
31                 if (card.suit == rectangleColor) {
32                     Value1.toInt(card.value) * 2  // Double the value if the suit matches
33                 } else if (rectangleColor == Suit.White) {
34                     Value1.toInt(card.value)  // Keep the value if placed on a white rectangle
35                 } else {
36                     0  // No points if the suit doesn't match
37                 }
38             case None => 0  // No card means no points
39         }
40     }
41 
42     // Method to display the grid
43     def display(): Unit = {
44         for (row <- grid) {
45             println(row.map {
46                 case Some(card) => s"${card.value} of ${card.suit}"
47                 case None       => "Empty"
48             }.mkString(" | "))
49         }
50     }
51 
52     // Method to return a string representation of the grid
53     def mesh(): String = {
54         grid.map { row =>
55             row.map {
56                 case Some(card) => s"${card.value} of ${card.suit}"
57                 case None       => "Empty"
58             }.mkString(" | ")
59         }.mkString(eol)
60     }
61 }
